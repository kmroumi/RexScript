app.rex 
// A simple Rex Script program
fn greet(name: str): str {
  give `Hello, {name}! 🦖`
}

fn main() {
  let users = ["Alice", "Bob", "Carol"]

  users
    |> filter(u => u.length > 3)
    |> map(greet)
    |> log

  loop i in 1..5 {
    log(`Count: {i}`)
  }
}

// Why Rex Script

Everything JS does.
Done better.

Rex Script is JavaScript under the hood — it compiles cleanly and runs anywhere JS runs. You keep your libraries, your ecosystem, and your knowledge.

⚡

Concise Syntax

New keywords like fn, give, and loop replace verbose JavaScript constructs. Write 30% less boilerplate.

🔒

Optional Typing

Add types when you need them. Skip them when you don't. Rex's gradual type system is inspired by TypeScript but lighter and built-in.

🔗

Pipeline Operator

Chain calls cleanly with |>. Data flows left to right — no more nested parentheses or temp variables.

🛡️

Null Safety

The exists keyword and ?/ null-coalesce operator make null checks first-class and readable.

📐

Range Literals

1..10 creates a range. Use it in loops, slices, and comparisons without importing a library.

🔄

100% JS Compatible

Rex compiles to clean, readable JavaScript. Use any npm package. Call Rex from JS, or JS from Rex. Zero lock-in.

// Syntax Changes

Familiar, but evolved.

If you know JavaScript, you already know most of Rex Script. The differences are small, deliberate, and always improve readability.

// JavaScript
function greet(name) {
  return "Hello, " + name;
}

const users = ["Alice", "Bob"];
users.forEach(function(u) {
  console.log(u);
});
// Rex Script
fn greet(name) {
  give "Hello, " + name
}

let users = ["Alice", "Bob"]
users |> forEach(u => log(u))
// JavaScript
for (let i = 0; i < 10; i++) {
  console.log(i);
}

const city = user
  && user.address
  && user.address.city
  || "Unknown";
// Rex Script
loop i in 0..9 {
  log(i)
}

let city = user?.address?.city
              ?/ "Unknown"
// JavaScript
if (x !== null && x !== undefined) {
  process(x);
}

try {
  throw new Error("oops");
} catch (err) {
  console.error(err);
}
// Rex Script
if x exists {
  process(x)
}

try! {
  fail "oops"
} catch err {
  log.error(err)
}

// Type System

Types, your way.

Rex Script's type system is fully optional. Annotate what matters, skip what doesn't. Types are inferred from context.

types.rex 
// Primitive types: str, num, bool, any, void
fn add(a: num, b: num): num {
  give a + b
}

// Type aliases
type UserId = str
type Point  = { x: num, y: num }

// Inferred — no annotation needed
let score = 42         // inferred: num
let name  = "Rex"     // inferred: str
let flag  = true      // inferred: bool

// Typed function with optional param
fn greet(name: str, loud?: bool): str {
  let msg = `Hello, {name}`
  give loud ? msg.upper() : msg
}

// Get Started

Install Rex Script

Get up and running in under a minute. Rex Script requires Node.js 18+.

1

Install the Rex compiler globally

Requires Node.js 18+ and npm.

npm install -g rex-script
2

Create your first file

Rex Script files use the .rex extension.

echo 'fn main() { give log("Hello!") }' > app.rex
3

Run it

Rex compiles and executes in one step.

rex run app.rex
4

Compile to JavaScript

Output clean, readable JS to deploy anywhere.

rex build app.rex --out dist/
